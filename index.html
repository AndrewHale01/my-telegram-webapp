#include #include using namespace std; #include int main() { double a, b, c; std::cout << "Введіть довжини трьох сторін трикутника: "; std::cin >> a >> b >> c; if (a <= 0 || b <= 0 || c <= 0) { std::cout << "Довжини сторін повинні бути додатними числами." << std::endl; return 1; } // Перевірка на існування трикутника if (a + b <= c || a + c <= b || b + c <= a) { std::cout << "Трикутник з такими сторонами не існує." << std::endl; return 1; } // Рівносторонній трикутник if (a == b && b == c && a == c) { std::cout << "Це рівносторонній трикутник." << std::endl; } // Рівнобедрений трикутник else if (a == b || a == c || b == c) { std::cout << "Це рівнобедрений трикутник." << std::endl; } // Перевірка на прямокутний трикутник (теорема Піфагора) // Сортуємо сторони, щоб c була гіпотенузою double sides[3] = { a, b, c }; std::sort(sides, sides + 3); // Використовуємо невеликий допуск для порівняння чисел з плаваючою комою if (std::abs(sides[0] * sides[0] + sides[1] * sides[1] - sides[2] * sides[2]) < 0.0001) { std::cout << "Це прямокутний трикутник." << std::endl; } // Якщо жоден з вищеперелічених if (!((a == b && b == c) || (a == b || a == c || b == c) || (std::abs(sides[0] * sides[0] + sides[1] * sides[1] - sides[2] * sides[2]) < 0.0001))) { std::cout << "Це різносторонній трикутник." << std::endl; } return 0; }
